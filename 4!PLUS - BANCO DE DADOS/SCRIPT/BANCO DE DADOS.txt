CREATE TABLE CLIENTE (
ID_CLIENTE INT PRIMARY KEY NOT NULL IDENTITY (1,1),
NOME VARCHAR (50)  NOT NULL,
CPF VARCHAR (20) NOT NULL UNIQUE,
CEP VARCHAR (20) NOT NULL,
ENDEREÇO VARCHAR(50),
NUMERO INT,
COMPLEMENTO VARCHAR(2),
BAIRRO VARCHAR(30),
CIDADE VARCHAR(30),
UF VARCHAR(2),
TEL_RESIDENCIAL VARCHAR(20),
TEL_CELULAR VARCHAR(20),
EMAIL VARCHAR(40),
)

CREATE TABLE VEICULO (
ID_VEICULO INT PRIMARY KEY NOT NULL IDENTITY (1,1),
ID_CLIENTE INT,
PLACA VARCHAR (20) NOT NULL,
MARCA VARCHAR(30),
MODELO VARCHAR(30),
ANO INT,
COR VARCHAR(20),
COMBUSTIVEL VARCHAR(20),
)

CREATE TABLE DEFEITO (

NOME_DEFEITO VARCHAR (50),
DESCRIÇÃO VARCHAR (200),

)

CREATE TABLE MECANICA (
ID_MECANICA INT PRIMARY KEY NOT NULL IDENTITY (1,1),
NOME_MECANICA VARCHAR(50)  NOT NULL,
CNPJ VARCHAR (20)  NOT NULL UNIQUE,
CEP VARCHAR(20) NOT NULL,
ENDEREÇO VARCHAR(50),
NUMERO INT,
COMPLEMENTO VARCHAR(2),
BAIRRO VARCHAR(30),
CIDADE VARCHAR(30),
UF VARCHAR(2),
TEL_COMERCIAL VARCHAR(20),
)

ID_OS INT PRIMARY KEY NOT NULL IDENTITY (1,1),
ID_CLIENTE INT,
ID_DEFEITO INT,
ID_VEICULO INT,
ID_MECANICA INT,
DATA VARCHAR(30),
STATUS_OS VARCHAR(30),
NOME_CLIENTE VARCHAR (50),
CPF_CLIENTE VARCHAR (20),
PLACA_CLIENTE VARCHAR (20),
MARCA_CLIENTE VARCHAR(30),
MODELO_CLIENTE VARCHAR(30),
ANO_CLIENTE INT,
NOME_MECANICA VARCHAR(50),
NOME_DEFEITO VARCHAR(50),
DESCRIÇÃO_DEFEITO VARCHAR (200),
VALOR_DEFEITO VARCHAR (20),

VALOR_TOTAL VARCHAR (20),
FOREIGN KEY(ID_CLIENTE) REFERENCES CLIENTE (ID_CLIENTE),
