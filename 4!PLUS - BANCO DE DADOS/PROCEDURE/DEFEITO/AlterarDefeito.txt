CREATE PROCEDURE uspDefeitoAlterar

		@ID_DEFEITO INT,
		@NOME_DEFEITO varchar (50),
        @DESCRIÇÃO varchar (200),
        @VALOR varchar (20)
        

AS
BEGIN

		UPDATE DEFEITO
		SET
       
           NOME_DEFEITO = @NOME_DEFEITO,
           DESCRIÇÃO = @DESCRIÇÃO,
           VALOR = @VALOR
           

		 WHERE

		 ID_DEFEITO = @ID_DEFEITO

	SELECT @ID_DEFEITO AS Retorno

END
