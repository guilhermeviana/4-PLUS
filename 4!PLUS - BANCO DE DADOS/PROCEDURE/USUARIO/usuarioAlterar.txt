CREATE PROCEDURE uspUsuarioAlterar

        @ID_USUARIO INT,
	@USUARIO VARCHAR (50),
	@SENHA VARCHAR (50),
	@REPITA_SENHA VARCHAR (50)
	

AS
BEGIN

UPDATE USUARIO
SET
	
           USUARIO = @USUARIO,
           SENHA = @SENHA,
           REPITA_SENHA = @REPITA_SENHA,
          
WHERE

ID_USUARIO = @ID_USUARIO

SELECT @ID_USUARIO AS Retorno

END
